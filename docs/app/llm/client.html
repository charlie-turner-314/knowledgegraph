120    def __init__(self, endpoint: Optional[str] = None, api_key: Optional[str] = None):
121        self.endpoint = endpoint or settings.llm_endpoint

125    @retry(wait=wait_random_exponential(multiplier=1, max=10), stop=stop_after_attempt(3))
126    def _call_api(self, payload: Dict[str, Any]) -> Dict[str, Any]:
127        if self.endpoint is None:
128            logger.error("LLM endpoint missing; skipping API call")
129            return {}
130        headers = {"Content-Type": "application/json"}
131        if self.api_key:
132            if settings.model_source == "azure":
133                headers["api-key"] = f"{self.api_key}"
134            else:
135                headers["Authorization"] = f"Bearer {self.api_key}"
136        response = requests.post(self.endpoint, headers=headers, data=orjson.dumps(payload))
137        print(response.status_code)
138        print(response.json())
139        exit()
140        response.raise_for_status()
141        return response.json()

505def _heuristic_parent_label(child_labels: Sequence[str]) -> str:
506    tokens_per_label = []
507    for label in child_labels:
508        tokens = {token for token in re.split(r"[^a-zA-Z0-9]+", label.lower()) if token}
509        tokens_per_label.append(tokens)
510    if not tokens_per_label:
511        return "Related Concept"
512
513    common_tokens = set.intersection(*tokens_per_label)
514    if common_tokens:
515        best = max(common_tokens, key=len)
516        return best.title()
517
518    # Fall back to the most frequent token across labels
519    frequency: Dict[str, int] = {}
520    for tokens in tokens_per_label:
521        for token in tokens:
522            frequency[token] = frequency.get(token, 0) + 1
523    if frequency:
524        best = max(frequency.items(), key=lambda item: (item[1], len(item[0])))[0]
525        return best.title()
526
527    return "Related Concept"
