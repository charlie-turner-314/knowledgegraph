 83    def delete_document(self, document: models.Document) -> None:
 84        """Remove a document and all dependent graph/candidate records."""
 85        # Delete edge sources tied to this document's chunks first to maintain referential integrity
 86        chunk_ids = [chunk.id for chunk in document.chunks]
 87        if chunk_ids:
 88            # Load edge sources via select to avoid hitting ORM cascades unexpectedly
 89            sources_stmt = select(models.EdgeSource).where(
 90                models.EdgeSource.document_chunk_id.in_(chunk_ids)
 91            )
 92            sources = list(self.session.exec(sources_stmt))
 93            edge_ids_to_check: set[int] = set()
 94            for source in sources:
 95                edge_ids_to_check.add(source.edge_id)
 96                self.session.delete(source)
 97
 98            if edge_ids_to_check:
 99                edges_stmt = select(models.Edge).where(models.Edge.id.in_(edge_ids_to_check))
100                edges = list(self.session.exec(edges_stmt))
101                for edge in edges:
102                    if not edge.sources:
103                        self.session.delete(edge)
104
105                # Remove nodes that no longer participate in any edges
106                orphan_nodes_stmt = select(models.Node).where(
107                    ~models.Node.outgoing_edges.any(),
108                    ~models.Node.incoming_edges.any(),
109                )
110                for node in self.session.exec(orphan_nodes_stmt):
111                    self.session.delete(node)
112
113        # Delete candidate triples tied to these chunks
114        if chunk_ids:
115            candidates_stmt = select(models.CandidateTriple).where(
116                models.CandidateTriple.chunk_id.in_(chunk_ids)
117            )
118            for candidate in self.session.exec(candidates_stmt):
119                self.session.delete(candidate)
120
121        # Delete chunks themselves
122        for chunk in document.chunks:
123            self.session.delete(chunk)
124
125        self.session.delete(document)

246        node: models.Node,
247        attributes: Sequence[Dict[str, object]],
248    ) -> None:
249        """Create or update the provided ``attributes`` for ``node``."""
250        if not attributes:
251            return
252        existing = {
253            attr.name.lower(): attr for attr in (node.attributes or [])
254        }
255
256        for payload in attributes:
257            name = str(payload.get("name", "")).strip()
258            if not name:
259                continue
260            key = name.lower()
261            data_type = payload.get("data_type", models.NodeAttributeType.string)
262            if isinstance(data_type, str):
263                try:
264                    data_type_enum = models.NodeAttributeType(data_type)
265                except ValueError:
266                    data_type_enum = models.NodeAttributeType.string
267            else:
268                data_type_enum = data_type
269
270            value_text = payload.get("value_text")
271            value_number = payload.get("value_number")
272            value_boolean = payload.get("value_boolean")
273
274            record = existing.get(key)
275            if record is None:
276                record = models.NodeAttribute(
277                    node_id=node.id,
278                    name=name,
279                    data_type=data_type_enum,
280                )
281                self.session.add(record)
282                if node.attributes is not None:
283                    node.attributes.append(record)
284            elif node.attributes is not None and record not in node.attributes:
285                node.attributes.append(record)
286            record.data_type = data_type_enum
287            record.value_text = value_text
288            record.value_number = value_number
289            record.value_boolean = value_boolean
290            record.updated_at = datetime.utcnow()

473    def _normalize_attributes(
474        self, attributes: Optional[Sequence[Dict[str, object]]]
475    ) -> List[Dict[str, object]]:
476        """Normalise external attribute payloads into a standard schema."""
477        if not attributes:
478            return []
479
480        normalized: Dict[str, Dict[str, object]] = {}
481        for raw in attributes:
482            if not isinstance(raw, dict):
483                continue
484            name_raw = raw.get("name") or raw.get("label")
485            if not name_raw:
486                continue
487            name = str(name_raw).strip()
488            if not name:
489                continue
490
491            data_type = raw.get("data_type")
492            if isinstance(data_type, models.NodeAttributeType):
493                data_type_enum = data_type
494            elif isinstance(data_type, str):
495                try:
496                    data_type_enum = models.NodeAttributeType(data_type)
497                except ValueError:
498                    data_type_enum = None
499            else:
500                data_type_enum = None
501
502            value_text = raw.get("value_text")
503            value_number = raw.get("value_number")
504            value_boolean = raw.get("value_boolean")
505
506            if value_text is None and value_number is None and value_boolean is None:
507                value = raw.get("value")
508                if isinstance(value, bool):
509                    data_type_enum = data_type_enum or models.NodeAttributeType.boolean
510                    value_boolean = value
511                elif isinstance(value, (int, float)):
512                    data_type_enum = data_type_enum or models.NodeAttributeType.number
513                    value_number = float(value)
514                elif value is not None:
515                    text_value = str(value).strip()
516                    if data_type_enum is None:
517                        lowered = text_value.lower()
518                        if lowered in {"true", "false", "yes", "no"}:
519                            data_type_enum = models.NodeAttributeType.boolean
520                            value_boolean = lowered in {"true", "yes"}
521                        else:
522                            try:
523                                numeric = float(text_value)
524                            except ValueError:
525                                if len(text_value.split()) <= 3 and lowered.replace("_", "").isalnum():
526                                    data_type_enum = models.NodeAttributeType.enum
527                                else:
528                                    data_type_enum = models.NodeAttributeType.string
529                                value_text = text_value
530                            else:
531                                data_type_enum = models.NodeAttributeType.number
532                                value_number = numeric
533                    else:
534                        value_text = text_value
535
536            if data_type_enum is None:
537                if value_number is not None:
538                    data_type_enum = models.NodeAttributeType.number
539                elif value_boolean is not None:
540                    data_type_enum = models.NodeAttributeType.boolean
541                else:
542                    data_type_enum = models.NodeAttributeType.string
543
544            if data_type_enum == models.NodeAttributeType.string and value_text is None:
545                fallback_value = raw.get("value")
546                if fallback_value is not None:
547                    value_text = str(fallback_value)
548
549            normalized[name.lower()] = {
550                "name": name,
551                "data_type": data_type_enum,
552                "value_text": value_text,
553                "value_number": value_number,
554                "value_boolean": value_boolean,
555            }
556
557        return list(normalized.values())

712class NodeEmbeddingStore:
713    """Shared in-memory embedding index for node labels."""
714
715    _BACKEND: str = "fuzzy"
716    _LABELS: List[str] = []
717    _LABEL_TO_INDEX: Dict[str, int] = {}
718    _VECTORS: Optional[np.ndarray] = None
719    _LOCK = threading.Lock()
720    _EMBEDDING_DIM: Optional[int] = None
721    _EXTERNAL_ENDPOINT: Optional[str] = None
722    _EXTERNAL_API_KEY: Optional[str] = None
723    _EXTERNAL_DEPLOYMENT: Optional[str] = None
724
725    def __init__(self, embedding_model: str = "external"):
726        """Initialise the similarity backend, preferring configured external embeddings."""
727        self.embedding_model = embedding_model
728        if settings.embedding_endpoint:
729            NodeEmbeddingStore._EXTERNAL_ENDPOINT = settings.embedding_endpoint
730            NodeEmbeddingStore._EXTERNAL_API_KEY = settings.embedding_api_key
731            NodeEmbeddingStore._EXTERNAL_DEPLOYMENT = settings.embedding_deployment
732            NodeEmbeddingStore._BACKEND = "external"
733        else:
734            NodeEmbeddingStore._BACKEND = "fuzzy"
735
736    @property
737    def backend(self) -> str:
738        """Return the current similarity backend (``external`` or ``fuzzy``)."""
739        return NodeEmbeddingStore._BACKEND
740
741    def has_entries(self) -> bool:
742        """Return ``True`` when at least one label has been cached."""
743        return bool(NodeEmbeddingStore._LABELS)
744
745    def encode_labels(self, labels: Sequence[str]) -> Optional[np.ndarray]:
746        """Encode ``labels`` using the configured external embedding endpoint."""
747        if NodeEmbeddingStore._BACKEND != "external":
748            return None
749        if not labels:
750            if NodeEmbeddingStore._EMBEDDING_DIM is None:
751                return np.empty((0, 0))
752            return np.empty((0, NodeEmbeddingStore._EMBEDDING_DIM))
753        return self._embed_external(labels)
754
755    def bulk_add(self, labels: Sequence[str]) -> None:
756        """Add ``labels`` to the similarity index if not already present."""
757        cleaned = [label.strip() for label in labels if label and isinstance(label, str) and label.strip()]
758        if not cleaned:
759            return
760        with NodeEmbeddingStore._LOCK:
761            to_add: List[str] = []
762            for label in cleaned:
763                key = label.lower()
764                if key in NodeEmbeddingStore._LABEL_TO_INDEX:
765                    continue
766                to_add.append(label)
767            if not to_add:
768                return
769
770            vectors: Optional[np.ndarray] = None
771            if NodeEmbeddingStore._BACKEND == "external":
772                vectors = self._embed_external(to_add)
773
774            for idx, label in enumerate(to_add):
775                key = label.lower()
776                NodeEmbeddingStore._LABEL_TO_INDEX[key] = len(NodeEmbeddingStore._LABELS)
777                NodeEmbeddingStore._LABELS.append(label)
778                if vectors is not None:
779                    vector = vectors[idx]
780                    if NodeEmbeddingStore._VECTORS is None:
781                        NodeEmbeddingStore._VECTORS = vector.reshape(1, -1)
782                    else:
783                        NodeEmbeddingStore._VECTORS = np.vstack([NodeEmbeddingStore._VECTORS, vector])
784
785    def suggest_similar(self, label: str, top_k: int = 5) -> List[Tuple[str, float]]:
786        """Return up to ``top_k`` labels similar to ``label`` using current backend."""
787        if not NodeEmbeddingStore._LABELS:
788            return []
789
790        label_key = label.lower()
791        results: List[Tuple[str, float]] = []
792        if (
793            NodeEmbeddingStore._BACKEND == "external"
794            and NodeEmbeddingStore._VECTORS is not None
795            and NodeEmbeddingStore._LABELS
796        ):
797            embedding = self._embed_external([label])
798            if embedding is None or embedding.size == 0:
799                logger.warning("External embedding call failed during query; falling back to fuzzy matching")
800            else:
801                query_vector = embedding[0]
802                similarities = NodeEmbeddingStore._VECTORS @ query_vector
803                order = np.argsort(-similarities)
804                for idx in order:
805                    candidate_label = NodeEmbeddingStore._LABELS[idx]
806                    if candidate_label.lower() == label_key:
807                        continue
808                    results.append((candidate_label, float(similarities[idx])))
809                    if len(results) >= top_k:
810                        break
811                return results
812
813        scores: List[Tuple[str, float]] = []
814        for candidate_label in NodeEmbeddingStore._LABELS:
815            if candidate_label.lower() == label_key:
816                continue
817            score = fuzz.token_set_ratio(label, candidate_label) / 100.0
818            if score > 0:
819                scores.append((candidate_label, float(score)))
820        scores.sort(key=lambda item: item[1], reverse=True)
821        return scores[:top_k]

823    def bootstrap_from_session(self, session: Session) -> None:
824        """Populate the store with existing node labels from the database."""
825        labels = [node.label for node in session.exec(select(models.Node)) if node.label]
826        self.bulk_add(labels)

828    def query(self, text: str, top_k: int = 8) -> List[Tuple[str, float]]:
829        """Return labels similar to ``text`` using embeddings or fuzzy fallback."""
830        if not NodeEmbeddingStore._LABELS:
831            return []
832
833        if (
834            NodeEmbeddingStore._BACKEND == "external"
835            and NodeEmbeddingStore._VECTORS is not None
836            and NodeEmbeddingStore._LABELS
837        ):
838            embedding = self._embed_external([text])
839            if embedding is not None and embedding.size:
840                query_vector = embedding[0]
841                similarities = NodeEmbeddingStore._VECTORS @ query_vector
842                order = np.argsort(-similarities)
843                results: List[Tuple[str, float]] = []
844                for idx in order[: top_k * 2]:
845                    candidate_label = NodeEmbeddingStore._LABELS[idx]
846                    score = float(similarities[idx])
847                    results.append((candidate_label, score))
848                return sorted(results, key=lambda item: item[1], reverse=True)[:top_k]
849            logger.warning("External embedding call failed; using fuzzy similarity for '%s'", text)
850
851        scores: List[Tuple[str, float]] = []
852        for candidate_label in NodeEmbeddingStore._LABELS:
853            score = fuzz.token_set_ratio(text, candidate_label) / 100.0
854            if score > 0:
855                scores.append((candidate_label, float(score)))
856        scores.sort(key=lambda item: item[1], reverse=True)
857        return scores[:top_k]

859    def _embed_external(self, texts: Sequence[str]) -> Optional[np.ndarray]:
860        """Call the configured embedding endpoint and return normalised vectors."""
861        if not texts:
862            return np.empty((0, NodeEmbeddingStore._EMBEDDING_DIM or 0))
863        if not NodeEmbeddingStore._EXTERNAL_ENDPOINT:
864            return None
865
866        payload: Dict[str, object] = {"input": list(texts)}
867        if NodeEmbeddingStore._EXTERNAL_DEPLOYMENT:
868            payload["model"] = NodeEmbeddingStore._EXTERNAL_DEPLOYMENT
869
870        headers = {"Content-Type": "application/json"}
871        if NodeEmbeddingStore._EXTERNAL_API_KEY:
872            headers["Authorization"] = f"Bearer {NodeEmbeddingStore._EXTERNAL_API_KEY}"
873            headers["api-key"] = NodeEmbeddingStore._EXTERNAL_API_KEY
874
875        try:
876            response = requests.post(
877                NodeEmbeddingStore._EXTERNAL_ENDPOINT,
878                headers=headers,
879                data=json.dumps(payload),
880                timeout=30,
881            )
882            response.raise_for_status()
883            body = response.json()
884        except Exception:  # noqa: BLE001
885            logger.exception("Embedding request failed; reverting to fuzzy similarity")
886            NodeEmbeddingStore._BACKEND = "fuzzy"
887            return None
888
889        vectors: List[Sequence[float]] = []
890        if isinstance(body, dict):
891            data_field = body.get("data")
892            if isinstance(data_field, list):
893                for item in data_field:
894                    if isinstance(item, dict):
895                        embedding = item.get("embedding") or item.get("vector")
896                        if embedding is not None:
897                            vectors.append(embedding)
898            elif isinstance(body.get("embeddings"), list):
899                vectors = body["embeddings"]
900        elif isinstance(body, list):
901            vectors = body
902
903        if not vectors:
904            logger.warning("Embedding endpoint returned no vectors; using fuzzy similarity")
905            return None
906        if len(vectors) != len(texts):
907            logger.warning(
908                "Embedding endpoint returned %s vectors for %s inputs",
909                len(vectors),
910                len(texts),
911            )
912
913        arr = np.array(vectors, dtype=float)
914        if arr.ndim != 2:
915            logger.warning("Embedding response has unexpected shape %s", arr.shape)
916            return None
917
918        norms = np.linalg.norm(arr, axis=1, keepdims=True)
919        norms[norms == 0] = 1.0
920        arr = arr / norms
921        NodeEmbeddingStore._EMBEDDING_DIM = arr.shape[1]
922        return arr

506

508                value = raw.get("value")
509                if isinstance(value, bool):
510                    data_type_enum = data_type_enum or models.NodeAttributeType.boolean
511                    value_boolean = value
512                elif isinstance(value, (int, float)):
513                    data_type_enum = data_type_enum or models.NodeAttributeType.number

523                            try:

531                            else:

539                    data_type_enum = models.NodeAttributeType.number
540                elif value_boolean is not None:
541                    data_type_enum = models.NodeAttributeType.boolean
542                else:

547                if fallback_value is not None:
