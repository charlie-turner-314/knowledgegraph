120    def ingest_text(
121        self,
122        *,
123        text: str,
124        title: Optional[str] = None,
125        media_type: str = "text/plain",
126    ) -> IngestionResult:
127        """Ingest raw ``text`` while reusing the structured extraction pipeline."""
128        cleaned = text.strip()
129        if not cleaned:
130            raise ValueError("Text input cannot be empty.")
131
132        checksum = hashlib.sha256(cleaned.encode("utf-8")).hexdigest()
133        parsed = ParsedDocument(
134            display_name=title.strip() if title and title.strip() else "Manual Entry",
135            media_type=media_type,
136            checksum=checksum,
137            chunks=[
138                ChunkPayload(
139                    ordering=0,
140                    text=cleaned,
141                    page_label="manual",
142                    token_count=len(cleaned.split()),
143                )
144            ],
145            original_path=None,
146            extra={"source": "manual_input"},
147        )
148        return self._ingest_parsed_document(parsed)
149
150    def _ingest_parsed_document(self, parsed: ParsedDocument) -> IngestionResult:
151        """Persist ``parsed`` content and extract candidate triples."""
152        existing_doc = self.documents.get_by_checksum(parsed.checksum)
153        was_existing = existing_doc is not None
154
155        if existing_doc:
156            document = existing_doc
157            if parsed.extra and not document.extra:
158                document.extra = parsed.extra
159            chunks_models = sorted(document.chunks or [], key=lambda chunk: chunk.ordering)
160            if not chunks_models:
161                chunks_models = self._persist_chunks(document, parsed)
162        else:
163            document = self.documents.create_document(
164                display_name=parsed.display_name,
165                media_type=parsed.media_type,
166                checksum=parsed.checksum,
167                original_path=str(parsed.original_path) if parsed.original_path else None,
168                extra=parsed.extra,
169            )
170            chunks_models = self._persist_chunks(document, parsed)
171
172        candidates = self._extract_candidates(chunks_models)
173
174        return IngestionResult(
175            document=document,
176            chunks=chunks_models,
177            candidate_triples=candidates,
178            was_existing_document=was_existing,
179        )
180
181    def _persist_chunks(
182        self, document: models.Document, parsed: ParsedDocument
183    ) -> List[models.DocumentChunk]:
184        """Store chunks that belong to ``document`` and return them."""
185        chunk_models = []
186        for payload in parsed.iter_chunks():
187            chunk_models.append(_chunk_to_model(document, payload))
188        return self.documents.add_chunks(document, chunk_models)
189
190    def _extract_candidates(
191        self, chunks: List[models.DocumentChunk]
192    ) -> List[models.CandidateTriple]:
193        """Extract candidates for each chunk while deduplicating duplicates."""
194        candidates: List[models.CandidateTriple] = []
195        seen_triples = set()
196        canonical_context = self._canonical_prompt_context()
197        for chunk in chunks:
198            response = self._call_extractor(chunk, canonical_context)
199            for triple in response.triples:
200                key = (
201                    triple.subject.strip(),
202                    triple.predicate.strip(),
203                    triple.object.strip(),
204                )
205                duplicate_of = self.candidates.find_similar(
206                    subject=key[0],
207                    predicate=key[1],
208                    obj=key[2],
209                )
210                is_duplicate = duplicate_of is not None or key in seen_triples
211                if is_duplicate:
212                    logger.info(f"Skipping duplicate triple: {key}")
213                    continue
214
215                # Check similarity to exising
216                similar_subjects = self.embedding_store.suggest_similar(key[0])
217                similar_objects = self.embedding_store.suggest_similar(key[2])
218
219
220                candidate = self._build_candidate(
221                    chunk,
222                    triple,
223                    response,
224                    duplicate_of=duplicate_of,
225                    is_duplicate=is_duplicate,
226                    similar_objects=similar_objects,
227                    similar_subjects=similar_subjects
228                )
229                self.session.add(candidate)
230                candidates.append(candidate)
231                seen_triples.add(key)
232        self.session.flush()
233        return candidates

235    def _call_extractor(
236        self,
237        chunk: models.DocumentChunk,
238        canonical_context: List[dict],
239    ) -> ExtractionResponse:
240        """Call the extraction LLM and return a validated response."""
241        metadata = {
242            "chunk_id": chunk.id,
243            "document_id": chunk.document_id,
244            "page_label": chunk.page_label,
245        }
246        try:
247            return self.llm_client.extract_triples(
248                chunk_text=chunk.text,
249                metadata=metadata,
250                canonical_context=canonical_context,
251            )
252        except Exception as exc:  # noqa: BLE001
253            logger.exception("LLM extraction failed for chunk %s", chunk.id)
254            # Return empty response preserving raw error for debugging
255            return ExtractionResponse(triples=[], raw_response={"error": str(exc)})

257    def _build_candidate(
258        self,
259        chunk: models.DocumentChunk,
260        triple: ExtractionTriple,
261        response: ExtractionResponse,
262        *,
263        duplicate_of: Optional[models.CandidateTriple],
264        is_duplicate: bool,
265        similar_subjects: list[tuple[str, float]],
266        similar_objects: list[tuple[str, float]]
267    ) -> models.CandidateTriple:
268        """Translate an ``ExtractionTriple`` into a candidate row with metadata."""
269        suggested_subject = None
270        suggested_object = None
271        if triple.suggested_subject_label:
272            suggested_subject = self.canonicals.find_best_match(triple.suggested_subject_label)
273        if triple.suggested_object_label:
274            suggested_object = self.canonicals.find_best_match(triple.suggested_object_label)
275
276        subject_attributes = _normalize_extracted_attributes(triple.subject_attributes)
277        object_attributes = _normalize_extracted_attributes(triple.object_attributes)
278        tags = sorted(
279            {
280                tag.strip().lower()
281                for tag in (triple.tags or [])
282                if isinstance(tag, str) and tag.strip()
283            }
284        )
285
286        return models.CandidateTriple(
287            chunk_id=chunk.id,
288            subject_text=triple.subject,
289            predicate_text=triple.predicate,
290            object_text=triple.object,
291            llm_confidence=triple.confidence,
292            llm_response_fragment=str(response.raw_response),
293            suggested_subject_term_id=suggested_subject.id if suggested_subject else None,
294            suggested_object_term_id=suggested_object.id if suggested_object else None,
295            duplicate_of_candidate_id=duplicate_of.id if duplicate_of else None,
296            is_potential_duplicate=is_duplicate,
297            subject_attributes=subject_attributes,
298            object_attributes=object_attributes,
299            suggested_tags=tags,
300        )

302    def _canonical_prompt_context(self) -> List[dict]:
303        """Return cached canonical vocabulary for LLM prompts."""
304        if self._canonical_context_cache is not None:
305            return self._canonical_context_cache
306        terms = self.canonicals.list_terms()
307        context: List[dict] = []
308        for term in terms:
309            payload = {
310                "label": term.label,
311                "aliases": term.aliases or [],
312            }
313            if term.entity_type:
314                payload["entity_type"] = term.entity_type
315            context.append(payload)
316        self._canonical_context_cache = context
317        return context

192    def _has_path(
193        self,
194        start: int,
195        goal: int,
196        adjacency: Dict[int, Set[int]],
197        *,
198        max_depth: int = 4,
199    ) -> bool:
200        """Breadth-first search to determine if two nodes connect within ``max_depth`` hops."""
201        if start == goal:
202            return True
203        visited: Set[int] = {start}
204        frontier: List[Tuple[int, int]] = [(start, 0)]
205        while frontier:
206            node_id, depth = frontier.pop(0)
207            if depth >= max_depth:
208                continue
209            for neighbor in adjacency.get(node_id, set()):
210                if neighbor == goal:
211                    return True
212                if neighbor in visited:
213                    continue
214                visited.add(neighbor)
215                frontier.append((neighbor, depth + 1))
216        return False

321    def _cluster_indices(
322        self,
323        similarity: np.ndarray,
324        threshold: float,
325        min_cluster_size: int,
326    ) -> List[List[int]]:
327        """Return connected components of indices above the similarity threshold."""
328        n = similarity.shape[0]
329        visited: Set[int] = set()
330        clusters: List[List[int]] = []
331        for idx in range(n):
332            if idx in visited:
333                continue
334            cluster = {idx}
335            stack = [idx]
336            while stack:
337                current = stack.pop()
338                for candidate in range(n):
339                    if candidate in cluster:
340                        continue
341                    if similarity[current, candidate] >= threshold:
342                        cluster.add(candidate)
343                        stack.append(candidate)
344            visited.update(cluster)
345            if len(cluster) >= min_cluster_size:
346                clusters.append(sorted(cluster))
347        return clusters

195    def _execute_plan(
196        self,
197        plan: QueryPlanResponse,
198        retrieval: QueryRetrievalResponse,
199    ) -> List[QueryMatch]:
200        """Traverse the graph according to ``plan`` and collect supporting triples."""
201        if not plan.plan or not plan.plan.steps:
202            return []
203
204        label_map = self._candidate_label_map(retrieval)
205        all_matches: Dict[int, QueryMatch] = {}
206
207        for step in plan.plan.steps:
208            start_label = step.start_node
209            node_ids = label_map.get(start_label.lower())
210            if not node_ids:
211                continue
212
213            predicates = [p.lower() for p in step.predicates] if step.predicates else []
214            directions = self._resolve_directions(step.direction)
215            depth_limit = max(1, min(step.depth or 1, plan.plan.max_hops or 3))
216
217            current_ids = set(node_ids)
218            visited_nodes: Set[int] = set(node_ids)
219            for hop in range(depth_limit):
220                edges = self._fetch_edges_for_step(current_ids, directions, predicates)
221                next_ids: Set[int] = set()
222                for edge in edges:
223                    match = self._to_query_match(edge)
224                    if match.edge_id is not None:
225                        all_matches.setdefault(match.edge_id, match)
226                    if "outbound" in directions and edge.subject_node_id in current_ids:
227                        if edge.object_node_id not in visited_nodes:
228                            next_ids.add(edge.object_node_id)
229                    if "inbound" in directions and edge.object_node_id in current_ids:
230                        if edge.subject_node_id not in visited_nodes:
231                            next_ids.add(edge.subject_node_id)
232                visited_nodes.update(next_ids)
233                current_ids = next_ids
234                if not current_ids:
235                    break
236
237        return list(all_matches.values())
